    # 创建项目

    # vue-cli 是安装3以前的，3之后的通过@vue/cli安装
yarn global add @vue/cli
# 更新vue cli相关模块的（以 @vue/cli-plugin- 或 vue-cli-plugin- 开头），执行 vue upgrade
yarn global upgrade --latest @vue/cli

vue create romantic
